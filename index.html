const select = document.getElementById('channel-select'); const video = document.getElementById('video-player'); const iframe = document.getElementById('iframe-player'); const placeholder = document.getElementById('placeholder'); const loader = document.getElementById('loading'); const liveStamp = document.getElementById('live-stamp'); const gridContainer = document.getElementById('channel-list-container'); // The new grid const searchInput = document.getElementById('channel-search'); // The new search bar const JSON_URL = './channels.json'; let channels = []; let hls = null; let shakaPlayer = null; // Stop any playing video/iframe (Logic Intact) async function stopAll() { if(loader) loader.style.display = 'none'; liveStamp.style.display = 'none'; if(hls){ hls.destroy(); hls = null; } if(shakaPlayer){ await shakaPlayer.destroy(); shakaPlayer = null; } video.pause(); video.removeAttribute('src'); video.load(); iframe.removeAttribute('src'); iframe.src = 'about:blank'; video.style.display = 'none'; iframe.style.display = 'none'; placeholder.style.display = 'flex'; } // Render both Dropdown (for compatibility) AND Grid Cards function renderUI(data){ channels = data; // 1. Keep original dropdown updated select.innerHTML = ''; // 2. Clear and Fill Grid gridContainer.innerHTML = ''; data.forEach((ch, i) => { // Dropdown logic const opt = document.createElement("option"); opt.value = i; opt.textContent = ch.name; select.appendChild(opt); // Grid Card logic const card = document.createElement('div'); card.className = 'channel-card'; card.setAttribute('data-name', ch.name.toLowerCase()); card.innerHTML = `
${ch.logo ? `` : `📺`}
${ch.name} `; card.onclick = () => { playChannel(i); window.scrollTo({top: 0, behavior: 'smooth'}); // Scroll to player on click }; gridContainer.appendChild(card); }); } // Search Filter Logic if(searchInput) { searchInput.oninput = () => { const term = searchInput.value.toLowerCase(); const cards = document.querySelectorAll('.channel-card'); cards.forEach(card => { const name = card.getAttribute('data-name'); card.style.display = name.includes(term) ? 'block' : 'none'; }); }; } // Load channels (Logic Intact) async function loadChannels(){ const cached = localStorage.getItem("channels_cache"); if(cached) renderUI(JSON.parse(cached)); try { const res = await fetch(JSON_URL, { cache: 'force-cache' }); const data = await res.json(); localStorage.setItem('channels_cache', JSON.stringify(data)); renderUI(data); } catch(e){ console.error('Channels load failed', e); } } // Play Logic (Strictly Intact as requested) async function playChannel(index){ const channel = channels[index]; if(!channel) return; await stopAll(); if(loader) loader.style.display = 'block'; placeholder.style.display = 'none'; liveStamp.style.display = channel.isLive ? 'block' : 'none'; if(channel.type === 'iframe'){ iframe.style.display = 'block'; iframe.src = channel.url; iframe.onload = () => { if(loader) loader.style.display = 'none'; }; return; } video.style.display = 'block'; const isHls = channel.type === 'hls' || channel.url.includes('.m3u8'); if(isHls){ if(window.Hls && Hls.isSupported()){ hls = new Hls({ enableWorker:true, lowLatencyMode:true, startLevel:-1 }); hls.loadSource(channel.url); hls.attachMedia(video); hls.on(Hls.Events.MANIFEST_PARSED, ()=> video.play()); } else if(video.canPlayType('application/vnd.apple.mpegurl')){ video.src = channel.url; video.play(); } return; } if(window.shaka && shaka.Player.isBrowserSupported()){ shaka.polyfill.installAll(); shakaPlayer = new shaka.Player(video); shakaPlayer.configure({ streaming:{ bufferingGoal:10, rebufferingGoal:0.5, lowLatencyMode:true } }); try{ await shakaPlayer.load(channel.url); video.play(); } catch(e){ console.error('Shaka error', e); if(loader) loader.style.display = 'none'; placeholder.style.display = 'flex'; } } } video.onplaying = () => { if(loader) loader.style.display = 'none'; }; select.onchange = () => { if(select.value !== '') playChannel(select.value); }; loadChannels();
